Quick Example

# Start the Kernel
import requests

# Initialize a session
session = requests.post("http://localhost:8000/runtime/init", json={
    "runtime": "langgraph",
    "config": {"mode": "strict"}
}).json()

# Execute with policy enforcement
result = requests.post(
    f"http://localhost:8000/runtime/{session['session_id']}/invoke",
    json={"action": "create_github_issue", "payload": {...}}
)
print(result.json())
